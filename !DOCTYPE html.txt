  document.getElementById('timerDisplay').className='timer-display running';
  document.getElementById('timerStatus').textContent='RUNNING';
  document.getElementById('timerLabel').textContent='● Recording focus session...';
  setRingColor('var(--accent3)');
  timerInterval=setInterval(()=>{timerSeconds++;document.getElementById('timerDisplay').textContent=fmtSecs(timerSeconds);updateRing(timerSeconds);},1000);
}
function pauseTimer(){
  if(!timerRunning)return;clearInterval(timerInterval);timerRunning=false;
  document.getElementById('timerDisplay').className='timer-display paused';
  document.getElementById('timerStatus').textContent='PAUSED';
  document.getElementById('timerLabel').textContent='⏸ Paused — click Start to resume';
  setRingColor('var(--yellow)');
}
function stopTimer(){
  if(timerSeconds===0){alert('Start a session first!');return;}
  clearInterval(timerInterval);timerRunning=false;
  const label=document.getElementById('sessionName').value.trim()||'Study Session';
  const xp=Math.max(1,Math.floor(timerSeconds/60)*2);
  timeLogs.unshift({id:Date.now(),label,duration:timerSeconds,
    startTime:sessionStart?sessionStart.toLocaleTimeString('en-IN',{hour:'2-digit',minute:'2-digit'}):new Date().toLocaleTimeString('en-IN',{hour:'2-digit',minute:'2-digit'}),
    xp,ts:nowISO()});
  save();renderLogs();renderScoreboard();renderHistory();updateDailyBar();
  if(window.pushMyScore) window.pushMyScore();
  resetTimer();document.getElementById('sessionName').value='';
  document.getElementById('timerLabel').textContent=`✓ Saved! +${xp} XP earned`;
  setTimeout(()=>{document.getElementById('timerLabel').textContent='Label your session below';},3000);
}
function resetTimer(){

  document.getElementById('timerStatus').textContent='READY';

function updateDailyBar(){
  const today=new Date().toISOString().split('T')[0];
  const secs=timeLogs.filter(l=>l.ts&&l.ts.startsWith(today)).reduce((a,l)=>a+l.duration,0);
  const pct=Math.min(100,(secs/14400*100)).toFixed(1);
  document.getElementById('focusBar').style.width=pct+'%';
  document.getElementById('focusPct').textContent=pct+'%';
  const mins=Math.floor(secs/60);
  const tips=['Start your first session!','Great start!','Building momentum!','Halfway there!','Almost at goal!','Daily goal crushed! 🏆'];
  document.getElementById('focusTip').textContent=`Today: ${mins}m focused — ${tips[Math.min(5,Math.floor(+pct/20))]}`;
}

function renderHistory(){
  const el=document.getElementById('historyList');
  if(currentTab==='sessions'){
    if(!timeLogs.length){el.innerHTML='<div class="empty-state">No sessions yet.</div>';return;}
    el.innerHTML=timeLogs.map((l,i)=>`
      <div class="log-item">
        <span class="sno">${i+1}</span><span class="log-time">${l.startTime}</span>
        <span class="log-label">${l.label}</span><span class="focus-dur">${fmtDur(l.duration)}</span>
        <span class="focus-tag">+${l.xp||0}XP</span><span class="log-date">${fmtDT(l.ts)}</span>
      </div>`).join('');
  } else if(currentTab==='tasks'){
    const done=todos.filter(t=>t.done);
    if(!done.length){el.innerHTML='<div class="empty-state">No completed tasks yet.</div>';return;}
    el.innerHTML=done.map((t,i)=>`
      <div class="todo-item done">
        <span class="sno" style="opacity:0.4">${i+1}</span>
        <div class="todo-check checked"></div>
        <span class="todo-text">${t.text}</span>
        ${t.dueDate?`<span class="todo-due">${fmtDate(t.dueDate)}</span>`:''}
        <span class="ts-small">Added: ${fmtDT(t.ts)}</span>
        ${t.doneAt?`<span class="ts-small" style="color:var(--accent3)">✓ ${fmtDT(t.doneAt)}</span>`:''}
        <span class="focus-tag">+20XP</span>
      </div>`).join('');
  } else {
    const s=[...backlog].sort((a,b)=>new Date(b.ts)-new Date(a.ts));
    if(!s.length){el.innerHTML='<div class="empty-state">No backlog history.</div>';return;}
    el.innerHTML=s.map((b,i)=>`
      <div class="backlog-item">
        <span class="sno">${i+1}</span>
        <span class="priority-badge p-${b.priority}">${b.priority.toUpperCase()}</span>
        <span class="backlog-text">${b.text}</span>
        <span class="ts-small" style="min-width:60px">${b.targetDate?fmtDate(b.targetDate):'—'}</span>
        <span class="ts-small">${fmtDT(b.ts)}</span>
      </div>`).join('');
  }
}
